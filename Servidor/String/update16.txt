ALTER TABLE IREGPESSOAJURIDICA ADD NATURESA VARCHAR(100);
ALTER TABLE IREGPESSOAJURIDICA ADD ORDEM INTEGER;
ALTER TABLE IREGPESSOAJURIDICA ADD TITULACAO VARCHAR(100);
ALTER TABLE IREGPESSOAJURIDICA ADD LEI VARCHAR(20);

CREATE VIEW VPROTOCOLOPJ(
  CODIGO,
  DATA,
  NOME,
  FICHA,
  NATURESA,
  ORDEM)
AS
SELECT
REGPESSOAJURIDICA.CODIGO,
REGPESSOAJURIDICA.DATA,
REGPESSOAJURIDICA.NOME,
REGPESSOAJURIDICA.FICHA,
IREGPESSOAJURIDICA.NATURESA,
IREGPESSOAJURIDICA.ORDEM
FROM REGPESSOAJURIDICA
INNER JOIN IREGPESSOAJURIDICA ON (IREGPESSOAJURIDICA.REGPESSOAJURIDICA = REGPESSOAJURIDICA.CODIGO)
;
